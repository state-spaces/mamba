PyTorch ANE — RMSNormANE

# RMSNormANE: no .float() cast, x*x avoids pow(2) CoreML regression
class RMSNormANE(nn.Module):
    def forward(self, x):
        variance = (x * x).mean(-1, keepdim=True)
        return x * torch.rsqrt(variance + self.eps) * self.weight

# Stage 3 — bias folded in here, not in RoPE kernel
K_norm     = self.B_norm(B_raw.view(1, 1, 1, self.d_state))
Q_norm     = self.C_norm(C_raw.view(1, 1, 1, self.d_state))
K_expanded = K_norm.repeat(1, 1, self.num_heads, 1).squeeze(1)
Q_expanded = Q_norm.repeat(1, 1, self.num_heads, 1).squeeze(1)
K_pre      = K_expanded + self.B_bias   # B_bias: (1, H, S)
Q_pre      = Q_expanded + self.C_bias
B_bias/C_bias are (1, H, S) — no rearrange needed. rsqrt() maps to a native ANE op. x*x instead of pow(2) avoids a CoreML power op that can force CPU fallback on some ANE firmware.

Angle Update + RoPE Pairwise⚡ full Triton kernel
apply_rotary_qk_inference_fwd — bias+angle+rotation fused; replace with apply_rope()

Two Fused Operations

[x'2i]= cos(θi)·x2i − sin(θi)·x2i+1
[x'2i+1]= sin(θi)·x2i + cos(θi)·x2i+1
Dims 32–63: pass through unchanged

# Bias was added in Stage 3 — only angle+rot here

                        .repeat(1, self.num_heads, 1)   # (1,H,16)

# apply_rope — stateless, operates on (1, H, S)
def apply_rope(self, x, cos, sin):
    x_rot  = x[..., :self.rotary_dim]          # (1,H,32)
    x_pass = x[..., self.rotary_dim:]
    x0, x1 = x_rot[..., 0::2], x_rot[..., 1::2]
    out_rot = torch.stack(
        [x0*cos - x1*sin, x0*sin + x1*cos], dim=-1
    ).flatten(-2)
    return torch.cat([out_rot, x_pass], dim=-1)
Input is (1,H,S) not (B,1,H,S) — bias was absorbed in Stage 3 so there is no squeeze needed here. The kernel used tl.sigmoid(2x)*2-1 as tanh approx; torch.tanh is numerically identical.

SSM Discretization Coefficients✓ safe
α, β, γ — all computed inside mamba3_step_fn, but trivial to extract

Trapezoid Discretization

αh= exp(Ah · Δth)  ←  decay
βh= (1 − λh) · Δth · αh  ←  past weight
γh= λh · Δth  ←  present weight
λ (trap) interpolates between ZOH (λ=0) and forward Euler (λ=1).

Tensor Shapes

A(1, 8)negative
DT(1, 8)positive
λ(1, 8)∈ (0, 1)
alpha, beta, gamma(1, 8)
g4, b4 (broadcast)(1, 8, 1, 1)

Original (inside CuteDSL mamba3_step_fn)

PyTorch ANE — explicit

# Compute all coefficients explicitly
alpha = torch.exp(A * DT)              # (1, H) — decay
beta  = (1.0 - lam) * DT * alpha     # (1, H) — past weight
gamma = lam * DT                       # (1, H) — present weight

# Broadcast to (1, H, 1, 1) for outer-product dims
g4 = gamma[:, :, None, None]
b4 = beta[:, :,  None, None]
torch.exp is ANE-safe. β reuses alpha — no redundant exp. [:,:,None,None] broadcast avoids einsum and maps to a reshape+mul on ANE.

# Reshape x_raw into (1, H, P) value vectors
V = x_raw.reshape(1, self.num_heads, self.headdim)     # (1, 8, 64)

# Outer products via broadcast matmul (avoids einsum — ANE-safe)
# (1,H,P,1) @ (1,H,1,S) → (1,H,P,S)
outer_curr = torch.matmul(V.unsqueeze(-1),
                           K_rot.unsqueeze(-2))           # (1, 8, 64, 64)
outer_prev = torch.matmul(self.v_state.unsqueeze(-1),
                           self.k_state.squeeze(1).unsqueeze(-2))

# Trapezoid state update
new_h = (alpha[:, :, None, None] * self.ssm_state
         + b4 * outer_prev + g4 * outer_curr)

# In-place register_buffer writes — picked up by ct.StateType export
self.ssm_state[:]   = new_h
self.k_state[:]     = K_rot.unsqueeze(1)    # store as (1, 1, 8, 64)
self.v_state[:]     = V
self.angle_state[:] = theta                   # written here, computed in Stage 4

Output Computation

Contract h (P×S matrix) with Q_rot (S vector):
yh= hh · Q_roth + Dh · Vh
   = Σs h[p,s] · Q_rot[s] + D · V[p]
D is the skip scalar — one per head.
SiLU gate (z) applied in Stage 8 at flat shape.

PyTorch ANE

# h·Q_rot: (1,H,P,S) @ (1,H,S,1) → squeeze → (1,H,P)
# Q_rot is already (1,H,S) — no squeeze(1) needed (not (1,1,H,S))

⚡ Original mamba3.py — step() [Triton + CuteDSL]
def step(self, u, angle_state, ssm_state, k_state, v_state):
    # u: (B, d_model=256)

    # 1. in_proj + split
    zxBCdt = self.in_proj(u)
    z, x, B, C, dd_dt, dd_A, trap, angles = \
        torch.split(zxBCdt, [
            self.d_inner, self.d_inner,
            self.d_state * self.num_bc_heads * self.mimo_rank,
            self.d_state * self.num_bc_heads * self.mimo_rank,
            self.nheads, self.nheads, self.nheads,
            self.num_rope_angles], dim=-1)

    # 2+3. Activations + KQ norm (calls _preprocess)
    DT, B, C, x, z, trap, A, angles = \
        self._preprocess(dd_A, dd_dt, B, C, x, z, trap, angles)
    # _preprocess uses: Triton RMSNormGated for B_norm/C_norm

    # 4. Bias + angle accumulation + RoPE
    bias_q = rearrange(self.C_bias, "h r n -> r h n")
    bias_k = rearrange(self.B_bias, "h r n -> r h n")
    rotate_pairwise = not self.is_mimo  # True for SISO
    C, B, nxt_angle_state = apply_rotary_qk_inference_fwd(
        q=C, k=B, angle_state=angle_state,
        angle_proj=angles, dt=DT,
        bias_q=bias_q, bias_k=bias_k,
        conjugate=False, inplace=False,
        rotate_pairwise=rotate_pairwise)
    # ↑ Triton kernel — NOT portable

    nxt_v_state = x
    nxt_k_state = B

    # SISO branch: xpj, zpj, outpj = ones
    xpj = torch.ones(self.mimo_rank, self.nheads,
                     self.headdim, device=x.device)
    zpj = torch.ones_like(xpj)
    outpj = torch.ones_like(xpj)

    # 5+6+7. SSM coefficients + state update + output
    y = torch.empty_like(x)
    mamba3_step_fn(
        ssm_state, k_state, v_state,
        A, B, C, self.D, x, DT, trap, xpj,
        outproj=outpj, state_out=None,
        out=y, z=z, zproj=zpj,
        tile_D=64, num_warps=4)
    # ↑ CuteDSL kernel — H100 only, NOT portable

    # 8. out_proj
    out = rearrange(y, "b h p -> b (h p)")
    out = self.out_proj(out.to(x.dtype))

    angle_state.copy_(nxt_angle_state)
    k_state.copy_(nxt_k_state)
    v_state.copy_(nxt_v_state)

    return out, nxt_angle_state, ssm_state, nxt_k_state, nxt_v_state
✓ MambaBlock.forward() — mamba_ane/modules/mamba3.py
def forward(self, x):
    # x: (1, d_model) — states live in register_buffers

    # 1. in_proj + explicit slice
    zxBCdt     = self.in_proj(x)
    z_raw      = zxBCdt[..., :self._s0]
    x_raw      = zxBCdt[..., self._s0:self._s1]
    B_raw      = zxBCdt[..., self._s1:self._s2]
    C_raw      = zxBCdt[..., self._s2:self._s3]
    dd_dt      = zxBCdt[..., self._s3:self._s4]
    dd_A       = zxBCdt[..., self._s4:self._s5]
    trap_raw   = zxBCdt[..., self._s5:self._s6]
    angles_raw = zxBCdt[..., self._s6:self._s7]

    # 2. Activations — no .float() casts
    A   = -(F.softplus(dd_A) + 1e-4)
    DT  = F.softplus(dd_dt + self.dt_bias)
    lam = torch.sigmoid(trap_raw)
    z   = F.silu(z_raw)                    # gate pre-activated

    # 3. K/Q RMSNormANE + expand + bias
    K_norm     = self.B_norm(B_raw.view(1, 1, 1, self.d_state))
    Q_norm     = self.C_norm(C_raw.view(1, 1, 1, self.d_state))
    K_pre      = K_norm.repeat(1, 1, self.num_heads, 1).squeeze(1) + self.B_bias
    Q_pre      = Q_norm.repeat(1, 1, self.num_heads, 1).squeeze(1) + self.C_bias

    # 4. Angle accumulation + RoPE (bias already in K_pre/Q_pre)
    angles      = angles_raw.view(1, 1, self.num_rope_angles).repeat(1, self.num_heads, 1)
    delta_theta = torch.tanh(angles) * math.pi * DT.unsqueeze(-1)
    theta       = self.angle_state + delta_theta
    K_rot       = self.apply_rope(K_pre, torch.cos(theta), torch.sin(theta))
    Q_rot       = self.apply_rope(Q_pre, torch.cos(theta), torch.sin(theta))

    # 5. Trapezoid coefficients
    alpha = torch.exp(A * DT)
    beta  = (1.0 - lam) * DT * alpha
    gamma = lam * DT
    g4    = gamma[:, :, None, None]
    b4    = beta[:, :,  None, None]

    # 6. SSM recurrence — matmul outer products; state writes
    V          = x_raw.reshape(1, self.num_heads, self.headdim)
    outer_curr = torch.matmul(V.unsqueeze(-1), K_rot.unsqueeze(-2))
    outer_prev = torch.matmul(self.v_state.unsqueeze(-1),
                               self.k_state.squeeze(1).unsqueeze(-2))
    new_h      = (alpha[:, :, None, None] * self.ssm_state
                  + b4 * outer_prev + g4 * outer_curr)
    self.ssm_state[:]   = new_h
    self.k_state[:]     = K_rot.unsqueeze(1)
    self.v_state[:]     = V
    self.angle_state[:] = theta

    # 7+8. Output + D skip + z gate + out_proj
    y_ssm = torch.matmul(new_h, Q_rot.unsqueeze(-1)).squeeze(-1)
    y_ssm = y_ssm + (V * self.D.view(1, -1, 1))
    return self.out_proj(y_ssm.reshape(1, self.d_inner) * z)
    # ↑ z gate applied flat; states in buffers — no tuple return
