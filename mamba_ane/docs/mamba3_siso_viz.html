# RMSNormANE (no .float() cast, x*x instead of pow(2))
class RMSNormANE(nn.Module):
    def forward(self, x):
        variance = (x * x).mean(-1, keepdim=True)
        rms_inv  = torch.rsqrt(variance + self.eps)
        return x * rms_inv * self.weight

# Stage 3 in forward()
K_norm     = self.B_norm(B_raw.view(1, 1, 1, self.d_state))       # (1, 1, 1, 64)
Q_norm     = self.C_norm(C_raw.view(1, 1, 1, self.d_state))
K_expanded = K_norm.repeat(1, 1, self.num_heads, 1).squeeze(1)  # (1, 8, 64)
Q_expanded = Q_norm.repeat(1, 1, self.num_heads, 1).squeeze(1)
K_pre      = K_expanded + self.B_bias                                # (1, 8, 64)
Q_pre      = Q_expanded + self.C_bias
RMSNormANE uses x*x (elementwise mul) instead of pow(2) because pow() with a literal exponent can produce a CoreML power op that forces CPU fallback on some ANE firmware versions. rsqrt() maps to a native ANE recip-sqrt op.

# apply_rope helper — stateless, called for both K and Q
def apply_rope(self, x, cos, sin):
    # x: (1, H, S=64), cos/sin: (1, H, 16)
    x_rot  = x[..., :self.rotary_dim]          # (1, H, 32)
    x_pass = x[..., self.rotary_dim:]          # (1, H, 32)
    x0 = x_rot[..., 0::2]                     # even dims  (1, H, 16)
    x1 = x_rot[..., 1::2]                     # odd dims   (1, H, 16)
    out_rot = torch.stack(
        [x0 * cos - x1 * sin,
         x0 * sin + x1 * cos], dim=-1).flatten(-2)  # (1, H, 32)
    return torch.cat([out_rot, x_pass], dim=-1)       # (1, H, 64)

# Stage 4 in forward()
angles      = angles_raw.view(1, 1, self.num_rope_angles).repeat(1, self.num_heads, 1)
delta_theta = torch.tanh(angles) * math.pi * DT.unsqueeze(-1)
theta       = self.angle_state + delta_theta
K_rot       = self.apply_rope(K_pre, torch.cos(theta), torch.sin(theta))
Q_rot       = self.apply_rope(Q_pre, torch.cos(theta), torch.sin(theta))
angle_state is a register_buffer — it is written in Stage 6 after the state update block. cos/sin computed per-step from accumulated theta; no precomputed table needed.

Trapezoid Discretization Coefficients✓ safe
α, β, γ — computed explicitly from A, DT, λ before the state update

Trapezoid Discretization

αh= exp(Ah · Δth)  ←  decay
βh= (1 − λh) · Δth · αh  ←  past weight
γh= λh · Δth  ←  present weight
λ=0 → ZOH (zero-order hold), λ=1 → forward Euler
Coefficients broadcast to (1,H,1,1) for outer-product dims

Tensor Shapes

A(1, 8)negative
DT(1, 8)positive
lam (λ)(1, 8)∈ (0,1)
alpha(1, 8)scalar
g4, b4 (broadcast)(1, 8, 1, 1)
alpha broadcast(1, 8, 1, 1)

ANE Implementation

alpha = torch.exp(A * DT)                   # (1, H=8)   — decay factor
beta  = (1.0 - lam) * DT * alpha            # (1, H=8)   — past weighting
gamma = lam * DT                             # (1, H=8)   — present weighting

# Broadcast to (1, H, 1, 1) for outer-product accumulation
g4 = gamma[:, :, None, None]              # (1, 8, 1, 1)
b4 = beta[:, :,  None, None]
torch.exp is ANE-safe. β = (1−λ)·Δt·α reuses alpha — no redundant exp. The [:,:,None,None] broadcast avoids einsum and maps to simple reshape+mul on ANE.

         + b4 * outer_prev

⬡ RMSNormANE — ANE-friendly norm
class RMSNormANE(nn.Module):
    """No .float() casts; x*x replaces pow(2)."""

    def __init__(self, dim, eps=1e-5):
        super().__init__()
        self.eps    = eps
        self.weight = nn.Parameter(torch.ones(dim))

    def forward(self, x):
        variance = (x * x).mean(-1, keepdim=True)
        rms_inv  = torch.rsqrt(variance + self.eps)
        return x * rms_inv * self.weight

def forward(self, x):
    # ── Stage 1: in_proj + explicit slice ──────────────────────────
    zxBCdt     = self.in_proj(x)
    z_raw      = zxBCdt[..., :self._s0]
    x_raw      = zxBCdt[..., self._s0:self._s1]
    B_raw      = zxBCdt[..., self._s1:self._s2]
    C_raw      = zxBCdt[..., self._s2:self._s3]
    dd_dt      = zxBCdt[..., self._s3:self._s4]
    dd_A       = zxBCdt[..., self._s4:self._s5]
    trap_raw   = zxBCdt[..., self._s5:self._s6]
    angles_raw = zxBCdt[..., self._s6:self._s7]

    # ── Stage 2: activations ──────────────────────────────────────
    A   = -(F.softplus(dd_A) + 1e-4)
    DT  = F.softplus(dd_dt + self.dt_bias)
    lam = torch.sigmoid(trap_raw)
    z   = F.silu(z_raw)

    # ── Stage 3: K/Q RMSNormANE + expand + bias ───────────────────
    K_norm     = self.B_norm(B_raw.view(1, 1, 1, self.d_state))
    Q_norm     = self.C_norm(C_raw.view(1, 1, 1, self.d_state))
    K_expanded = K_norm.repeat(1, 1, self.num_heads, 1).squeeze(1)
    Q_expanded = Q_norm.repeat(1, 1, self.num_heads, 1).squeeze(1)
    K_pre      = K_expanded + self.B_bias                           # (1, H, S)
    Q_pre      = Q_expanded + self.C_bias

    # ── Stage 4: RoPE angle accumulation ──────────────────────────
    angles      = angles_raw.view(1, 1, self.num_rope_angles).repeat(1, self.num_heads, 1)
    delta_theta = torch.tanh(angles) * math.pi * DT.unsqueeze(-1)
    theta       = self.angle_state + delta_theta
    K_rot       = self.apply_rope(K_pre, torch.cos(theta), torch.sin(theta))
    Q_rot       = self.apply_rope(Q_pre, torch.cos(theta), torch.sin(theta))

    # ── Stage 5: trapezoid coefficients ───────────────────────────
    alpha = torch.exp(A * DT)
    beta  = (1.0 - lam) * DT * alpha
    gamma = lam * DT
    g4    = gamma[:, :, None, None]
    b4    = beta[:, :,  None, None]

    # ── Stage 6: SSM recurrence + state writes ────────────────────
    V           = x_raw.reshape(1, self.num_heads, self.headdim)
    outer_curr  = torch.matmul(V.unsqueeze(-1),
                               K_rot.unsqueeze(-2))
    outer_prev  = torch.matmul(self.v_state.unsqueeze(-1),
                               self.k_state.squeeze(1).unsqueeze(-2))
    new_h       = (alpha[:, :, None, None] * self.ssm_state
                   + b4 * outer_prev
                   + g4 * outer_curr)

    self.ssm_state[:]   = new_h
    self.k_state[:]     = K_rot.unsqueeze(1)
    self.v_state[:]     = V
    self.angle_state[:] = theta

    # ── Stages 7+8: output contraction + D skip + out_proj ────────
    y_ssm = torch.matmul(new_h, Q_rot.unsqueeze(-1)).squeeze(-1)
    y_ssm = y_ssm + (V * self.D.view(1, -1, 1))
    return self.out_proj(y_ssm.reshape(1, self.d_inner) * z)
